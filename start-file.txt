#  Open your browser:
        http://localhost:3000 → Your Next.js app
        http://localhost:8000 → Django welcome page (or your API)
        http://localhost:8000/admin → Django admin (create a superuser if needed)

#  Create a superuser (optional, one time):

        docker-compose exec backend python manage.py createsuperuser

#  Your daily commands (bookmark these):

        # Start everything (normal mode)
        docker-compose up

        # Start in background
        docker-compose up -d

        # Stop everything
        docker-compose down

        # Full clean restart (when you change code/structure)
        docker-compose down --remove-orphans && docker-compose up
